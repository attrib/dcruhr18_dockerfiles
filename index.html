docker-compose.yml


              version: '3'
              services:
                web:
                  build: .
                  depends_on:
                    - db
                    - solr
                    - redis
                    - selenium
                db:
                  image: mariadb
                solr:
                  image: solr
                redis:
                  image: redis
                selenium:
                  image: selenium/standalone-chrome:3.8.1-bohrium
                  volumes:
                    - /dev/shm:/dev/shm
